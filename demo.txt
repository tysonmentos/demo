**************************************Shell Questions******************************************

1. Write a menu based script which displays the following options :
* Accept roll number, name, marks in sub1, sub2 and sub3. Calculate total, percentage, grade & class and enter the record in a file “student”. The marks are out of 100 in each subject. Allow the user to enter any number of records.


*                    Grade is found out as follows :
*                 Percentage                 Grade
*                 < 35                                Fail
*                 >= 35 & < 50                Third
*                 >= 50 & < 60                Second
*                 >= 60 & < 75                First
*                 >= 75                         Distn

*                     If the student secures < 35 in any one of the subjects, then class = “fail”.         Otherwise     
*                         class=”pass”.



----------Answer--------------

#!/bin/bash


# Function to calculate grade
calculate_grade() {
    if (( $1 < 35 )); then
        grade="Fail"
    elif (( $1 < 50 )); then
        grade="Third"
    elif (( $1 < 60 )); then
        grade="Second"
    elif (( $1 < 75 )); then
        grade="First"
    else
        grade="Distn"
    fi
}


# Function to calculate class
calculate_class() {
    if (( $1 < 35 || $2 < 35 || $3 < 35 )); then
        class="Fail"
    else
        class="Pass"
    fi
}


# Function to display menu
display_menu() {
    echo "==============================="
    echo "      STUDENT MENU"
    echo "==============================="
    echo "1. Enter Student Record"
    echo "2. Display Student Records"
    echo "3. Exit"
    echo "==============================="
}


# Function to enter student record
enter_record() {
    read -p "Enter Roll Number: " roll_number
    read -p "Enter Name: " name
    read -p "Enter Marks in Subject 1: " sub1
    read -p "Enter Marks in Subject 2: " sub2
    read -p "Enter Marks in Subject 3: " sub3


    total=$((sub1 + sub2 + sub3))
    percentage=$((total / 3))


    calculate_grade $percentage
    calculate_class $sub1 $sub2 $sub3


    echo "$roll_number | $name | $sub1 | $sub2 | $sub3 | $total | $percentage | $grade | $class" >> student


    echo "Record added successfully."
}


# Function to display student records
display_records() {
    if [ -f student ]; then
        echo
        echo "Roll | Name | Sub1 | Sub2 | Sub3 | Total | % | Grade | Class"
        echo "--------------------------------------------------------------"
        cat student
    else
        echo "No records found."
    fi
}


# Main Program
while true
do
    display_menu
    read -p "Enter your choice: " choice


    case $choice in
        1) enter_record ;;
        2) display_records ;;
        3) echo "Exiting Program..."
           exit ;;
        *) echo "Invalid Choice!" ;;
    esac
done

* Write a bash script to take input string from user.  Print output as follows. 
*         a) If all the characters in input string is upper case, print “Input String is in UPPER case”
*         b) If all the characters in input string is lower case, print “Input String is in LOWER case”
*         c) else print “Input String is in MIXED case”

-------------------------Answer---------------------------
#!/bin/bash
# Accept input string from user
echo "Enter a string:"
read str


# Check if the string is all uppercase
if [[ "$str" == "${str^^}" && "$str" != "${str,,}" ]]; then
    echo "Input String is in UPPER case"


# Check if the string is all lowercase
elif [[ "$str" == "${str,,}" && "$str" != "${str^^}" ]]; then
    echo "Input String is in LOWER case"


# Otherwise, it is mixed case
else
    echo "Input String is in MIXED case"
Fi

* Write a bash script to take input Number as upper limit from user.  
*         a) Print the PID of the current script
*         b) Print the Numbers in reverse till we reach zero 
*                 e.g: If the user enters 5, print the numbers as below in separate lines
*                         5
*                         4
*                         3
*                         2
*                         1 
*         c) Input Number range is greater than 1 and less than 100
*         d) If user inputs number less than 1 or more than 100 print message “Input Number is Out of range”


-------------------------Answer---------------------------

#!/bin/bash
# Print the PID of the current script
echo "Current Script PID: $$"
# Read the upper limit
echo "Enter a number (2-99):"
read num


# Check the range
if [ $num -le 1 ] || [ $num -ge 100 ]; then
    echo "Input Number is Out of range"
else
    echo "Numbers in reverse:"
    while [ $num -gt 0 ]
    do
        echo $num
        num=$((num-1))
    done
Fi

* Write a bash script to take two input string from user. Print the output as follows 
*         a) If the input string 1 and input String 2 matches, Print “The TWO STRING matches”. The match should not be case sensitive. Means Input String 1 is  “SPARK” and input 2 string is “Spark”, still it should print “The TWO STRING matches”.
*         b) If the input string 1 and input String 2 does not match, Print “The TWO STRING DOES NOT match”

-------------------------Answer---------------------------

#!/bin/bash
# Read first string
echo "Enter First String:"
read str1


# Read second string
echo "Enter Second String:"
read str2


# Compare strings (case-insensitive)
if [ "${str1,,}" = "${str2,,}" ]; then
    echo "The TWO STRING matches"
else
    echo "The TWO STRING DOES NOT match"
Fi

********************************Permission based commands**********************************


1. Give all permissions (Read, Write, Execute) to the owner
chmod u+rwx filename

2. Give execute permission to the owner
chmod u+x filename

3. Give read permission
chmod u+r filename

4. Give write permission
chmod u+w filename

5. Give permissions to everyone (owner, group, others)
chmod a+rwx filename




********************************Numeric Permission based commands**********************************


chmod 777 filename
	Read, Write, Execute for everyone

chmod 755 filename
	Owner: rwx, Group: r-x, Others: r-x

chmod 744 filename
	Owner: rwx, Group: r--, Others: r--

chmod 700 filename
	Owner: rwx only

chmod 644 filename
	Owner: rw-, Group: r--, Others: r--
	
Check current permissions
ls -l filename
